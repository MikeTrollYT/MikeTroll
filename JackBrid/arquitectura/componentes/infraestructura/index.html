      context: ./backend
      dockerfile: Dockerfile

      - JACKETT_URL=http://jackett:9117
      - JACKETT_API_KEY=${JACKETT_API_KEY}
      - ALLDEBRID_API_KEY=${ALLDEBRID_API_KEY}
    ports:
      - "3000:3000"
    depends_on:
      - jackett
    restart: unless-stopped
    networks:
      - jackbrid-network

  # Nginx - Frontend + Reverse Proxy
  nginx:
    build:
      context: .
      dockerfile: ./nginx/Dockerfile
    container_name: jackbrid-nginx
    ports:
      - "8998:80"
    depends_on:
      - backend
    restart: unless-stopped
    networks:
      - jackbrid-network

networks:
  jackbrid-network:
    driver: bridge
