Polling Inteligente¶

class AllDebridMonitor {
  constructor() {
    this.items = new Map();
    this.interval = null;
  }

  addItem(magnetId) {
    this.items.set(magnetId, { statusCode: 0, lastCheck: Date.now() });
    this.startPolling();
  }

  removeItem(magnetId) {
    this.items.delete(magnetId);
    if (this.items.size === 0) {
      this.stopPolling();
    }
  }

  startPolling() {
    if (this.interval) return;

    this.interval = setInterval(async () => {
      const ids = Array.from(this.items.keys());
      const response = await fetch('/status', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ ids })
      });

      const data = await response.json();

      ids.forEach(id => {
        const status = data.statusById[id];
        const item = this.items.get(id);

        // Actualizar UI
        this.updateUI(id, status);

        // Si est谩 listo o error, dejar de monitorear
        if (status.statusCode === 4 || status.statusCode >= 5) {
          this.removeItem(id);
        }
      });
    }, 10000); // 10 segundos
  }

  stopPolling() {
    if (this.interval) {
      clearInterval(this.interval);
      this.interval = null;
    }
  }

  updateUI(id, status) {
    const element = document.getElementById(`item-${id}`);
    if (!element) return;

    element.querySelector('.status').textContent = status.status;
    element.querySelector('.progress').style.width = `${status.processingPerc}%`;

    if (status.downloadSpeed > 0) {
      const speedMB = (status.downloadSpeed / 1048576).toFixed(2);
      element.querySelector('.speed').textContent = `${speedMB} MB/s`;
    }
  }
}

async function reproduceVideo(magnetId) {
  // 1. Verificar que est茅 listo
  const response = await fetch('/status', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ ids: [magnetId] })
  });

  const statusData = await response.json();
  const status = statusData.statusById[magnetId];

  if (status.statusCode !== 4) {
    alert(`El video a煤n no est谩 listo. Estado: ${status.status}`);
    return;
  }

  // 2. Obtener links
  const linksResponse = await fetch('/download-links', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ id: magnetId })
  });

  const linksData = await linksResponse.json();

  if (linksData.links.length === 0) {
    alert('No se encontraron archivos de video');
    return;
  }

  // 3. Abrir reproductor

  openPlayer(video.link, video.filename);
}
