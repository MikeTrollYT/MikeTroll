Módulos del servidor backend de Jackent.

app.js¶

Ubicación: /backend/app.js

Responsabilidad: Servidor Express principal

Estructura:

const express = require('express');
const cors = require('cors');
const jackett = require('./jackettClient');
const qbittorrent = require('./qbittorrentClient');

const app = express();
const PORT = 3000;

// Middleware
app.use(cors());
app.use(express.json({ limit: '5mb' }));

// Routes
app.get('/health', async (req, res) => { /* ... */ });
app.get('/trackers', async (req, res) => { /* ... */ });
app.get('/search', async (req, res) => { /* ... */ });
app.post('/add', async (req, res) => { /* ... */ });
app.get('/torrents', async (req, res) => { /* ... */ });
app.post('/delete', async (req, res) => { /* ... */ });
app.get('/config', async (req, res) => { /* ... */ });
app.post('/config', async (req, res) => { /* ... */ });

// Start server
app.listen(PORT, () => {
  console.log(`✅ Backend listening on port ${PORT}`);
});

Manejo de errores:

// Try-catch en cada endpoint
try {
  // Lógica
} catch (err) {
  console.error('Error:', err.message);
  res.status(500).json({ error: err.message });
}

jackettClient.js¶

Ubicación: /backend/jackettClient.js

Responsabilidad: Cliente para API de Jackett

Implementación:

const fetch = require('node-fetch');

const JACKETT_URL = process.env.JACKETT_URL || 'http://jackett:9117';
const API_KEY = process.env.JACKETT_API_KEY;

/**
 * Obtiene lista de indexers configurados
 */
async function getIndexers() {
  const url = `${JACKETT_URL}/api/v2.0/indexers?apikey=${API_KEY}`;
  const response = await fetch(url);

  if (!response.ok) {
    throw new Error('Failed to fetch indexers from Jackett');
  }

  const data = await response.json();
  return data.map(indexer => ({
    id: indexer.id,
    name: indexer.name,
    type: indexer.type,
    language: indexer.language,
    configured: indexer.configured
  }));
}

qbittorrentClient.js¶

Ubicación: /backend/qbittorrentClient.js

Responsabilidad: Cliente para qBittorrent Web API

Implementación:

const fetch = require('node-fetch');

const QBITTORRENT_URL = process.env.QBITTORRENT_URL || 'http://qbittorrent:8080';
const QBITTORRENT_USER = process.env.QBITTORRENT_USER || 'admin';
const QBITTORRENT_PASSWORD = process.env.QBITTORRENT_PASSWORD || 'adminadmin';

let cookie = null;

/**
 * Autenticación con qBittorrent
 */
async function login() {
  const response = await fetch(`${QBITTORRENT_URL}/api/v2/auth/login`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/x-www-form-urlencoded'
    },
    body: `username=${QBITTORRENT_USER}&password=${QBITTORRENT_PASSWORD}`
  });

  if (response.ok) {
    cookie = response.headers.get('set-cookie');
    return true;
  }
  throw new Error('Failed to login to qBittorrent');
}

/**
 * Añade magnet o torrent
 */
async function addTorrent(magnetUrl) {
  if (!cookie) await login();

  const params = new URLSearchParams();
  params.append('urls', magnetUrl);
  params.append('savepath', '/downloads');

  const response = await fetch(`${QBITTORRENT_URL}/api/v2/torrents/add`, {
    method: 'POST',
    headers: {
      'Cookie': cookie,
      'Content-Type': 'application/x-www-form-urlencoded'
    },
    body: params
  });

  if (response.ok) {
    const text = await response.text();
    return { success: text === 'Ok.' };
  }
  throw new Error('Failed to add torrent');
}

/**
 * Lista todos los torrents
 */
async function listTorrents() {
  if (!cookie) await login();

  const response = await fetch(`${QBITTORRENT_URL}/api/v2/torrents/info`, {
    headers: { 'Cookie': cookie }
  });

  if (response.ok) {
    return await response.json();
  }
  throw new Error('Failed to list torrents');
}

/**
 * Elimina un torrent
 */
async function deleteTorrent(hash, deleteFiles = true) {
  if (!cookie) await login();

  const params = new URLSearchParams();
  params.append('hashes', hash);
  params.append('deleteFiles', deleteFiles);

  const response = await fetch(`${QBITTORRENT_URL}/api/v2/torrents/delete`, {
    method: 'POST',
    headers: {
      'Cookie': cookie,
      'Content-Type': 'application/x-www-form-urlencoded'
    },
    body: params
  });

  return response.ok;
}

/**
 * Obtiene propiedades de un torrent
 */
async function getTorrentProperties(hash) {
  if (!cookie) await login();

  const response = await fetch(
    `${QBITTORRENT_URL}/api/v2/torrents/properties?hash=${hash}`,
    { headers: { 'Cookie': cookie } }
  );

  if (response.ok) {
    return await response.json();
  }
  throw new Error('Failed to get torrent properties');
}

/**
 * Obtiene archivos de un torrent
 */
async function getTorrentFiles(hash) {
  if (!cookie) await login();

  const response = await fetch(
    `${QBITTORRENT_URL}/api/v2/torrents/files?hash=${hash}`,
    { headers: { 'Cookie': cookie } }
  );

  if (response.ok) {
    return await response.json();
  }
  throw new Error('Failed to get torrent files');
}

module.exports = {
  login,
  addTorrent,
  listTorrents,
  deleteTorrent,
  getTorrentProperties,
  getTorrentFiles
};
