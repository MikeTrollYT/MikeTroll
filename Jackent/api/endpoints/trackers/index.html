Uso en el Frontend¶

Renderizar Lista de Trackers¶

async function loadTrackers() {
  const response = await fetch('/api/trackers');
  const data = await response.json();

  const trackerList = document.getElementById('tracker-list');
  trackerList.innerHTML = '';

  data.trackers.forEach(tracker => {
    const checkbox = document.createElement('input');
    checkbox.type = 'checkbox';
    checkbox.id = tracker.id;
    checkbox.value = tracker.id;
    checkbox.disabled = !tracker.configured;

    const label = document.createElement('label');
    label.htmlFor = tracker.id;
    label.textContent = tracker.name;

    if (!tracker.configured) {
      label.classList.add('disabled');
      label.title = tracker.last_error || 'Not configured';
    }

    trackerList.appendChild(checkbox);
    trackerList.appendChild(label);
  });
}
