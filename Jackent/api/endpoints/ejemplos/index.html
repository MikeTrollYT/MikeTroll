class JackentAPI {
  constructor(baseURL = '/api') {
    this.baseURL = baseURL;
  }

  async search(query, options = {}) {
    const params = new URLSearchParams({
      q: query,
      trackers: options.trackers || '1337x,thepiratebay',
      sort: options.sort || 'seeders',
      limit: options.limit || 50,
      onlySeeded: options.onlySeeded ? 'yes' : 'no'
    });

    const response = await fetch(`${this.baseURL}/search?${params}`);
    return await response.json();
  }

    const response = await fetch(`${this.baseURL}/add`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ raw: torrent.raw })
    });
    return await response.json();
  }

  async listDownloads() {
    const response = await fetch(`${this.baseURL}/list`);
    return await response.json();
  }

  async pauseTorrent(torrentHash) {
    const response = await fetch(`${this.baseURL}/pause`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: torrentHash })
    });
    return await response.json();
  }

  async resumeTorrent(torrentHash) {
    const response = await fetch(`${this.baseURL}/resume`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: torrentHash })
    });
    return await response.json();
  }

  async deleteTorrent(torrentHash, deleteFiles = true) {
    const response = await fetch(`${this.baseURL}/delete`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },

  }
}

import requests
from urllib.parse import urlencode
from typing import List, Dict, Optional

class JackentClient:
    def __init__(self, base_url: str = "http://localhost:7997"):
        self.base_url = base_url
        self.session = requests.Session()

    def search(
        self,
        query: str,
        trackers: List[str] = None,
        sort: str = 'seeders',
        limit: Optional[int] = None,
        only_seeded: bool = True
    ) -> Dict:
        if trackers is None:
            trackers = ['1337x', 'thepiratebay']

        params = {
            'q': query,
            'trackers': ','.join(trackers),
            'sort': sort,
            'onlySeeded': 'yes' if only_seeded else 'no'
        }

        if limit:
            params['limit'] = limit

        response = self.session.get(f"{self.base_url}/search", params=params)
        response.raise_for_status()
        return response.json()

    def add_torrent(self, torrent_raw: Dict) -> Dict:
        payload = {'raw': torrent_raw}
        response = self.session.post(
            f"{self.base_url}/add",
            json=payload
        )
        response.raise_for_status()
        return response.json()

    def list_downloads(self) -> List[Dict]:
        response = self.session.get(f"{self.base_url}/list")
        response.raise_for_status()
        return response.json()['items']

    def pause_torrent(self, torrent_hash: str) -> bool:
        response = self.session.post(
            f"{self.base_url}/pause",
            json={'id': torrent_hash}
        )
        response.raise_for_status()
        return response.json()['success']

    def resume_torrent(self, torrent_hash: str) -> bool:
        response = self.session.post(
            f"{self.base_url}/resume",
            json={'id': torrent_hash}
        )
        response.raise_for_status()
        return response.json()['success']

    def delete_torrent(self, torrent_hash: str, delete_files: bool = True) -> bool:
        response = self.session.post(
            f"{self.base_url}/delete",
            json={'id': torrent_hash, 'deleteFiles': delete_files}
        )
        response.raise_for_status()
        return response.json()['success']

const fetch = require('node-fetch');

class JackentAPI {
  constructor(baseURL) {
    this.baseURL = baseURL;
  }

  async request(endpoint, options = {}) {
    const url = `${this.baseURL}${endpoint}`;
    const response = await fetch(url, options);

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }

    return await response.json();
  }

  async search(query, options = {}) {
    const params = new URLSearchParams({
      q: query,
      trackers: options.trackers || '1337x',
      sort: options.sort || 'seeders',
      ...(options.limit && { limit: options.limit }),
      onlySeeded: options.onlySeeded ? 'yes' : 'no'
    });

    return await this.request(`/search?${params}`);
  }

  async addTorrent(torrentRaw) {
    return await this.request('/add', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ raw: torrentRaw })
    });
  }

  async listDownloads() {
    return await this.request('/list');
  }

  async pauseTorrent(torrentHash) {
    return await this.request('/pause', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: torrentHash })
    });
  }

  async resumeTorrent(torrentHash) {
    return await this.request('/resume', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: torrentHash })
    });
  }

  async deleteTorrent(torrentHash, deleteFiles = true) {
    return await this.request('/delete', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ id: torrentHash, deleteFiles })
    });
  }

  async waitUntilComplete(torrentHash, maxWait = 3600000, interval = 10000) {
    const startTime = Date.now();

    while (Date.now() - startTime < maxWait) {
      const downloads = await this.listDownloads();
      const torrent = downloads.items.find(t => t.id === torrentHash);

      if (!torrent) {
        throw new Error('Torrent not found');
      }

      if (torrent.progress >= 1.0) {
        return true; // Complete
      }

      await new Promise(resolve => setTimeout(resolve, interval));
    }

    throw new Error('Timeout waiting for download to complete');
  }
}

const api = new JackentAPI('http://localhost:7997');

async function downloadTorrent(query) {
  // Buscar
  const results = await api.search(query, { 
    trackers: '1337x',

  console.log('Torrent a帽adido a qBittorrent');

    const downloads = await api.listDownloads();

import { useState, useEffect, useCallback } from 'react';

function useJackent(baseURL = '/api') {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const search = useCallback(async (query, options = {}) => {
    setLoading(true);
    setError(null);

    try {
      const params = new URLSearchParams({
        q: query,
        trackers: options.trackers || '1337x,thepiratebay',
        sort: options.sort || 'seeders',
        ...(options.limit && { limit: options.limit }),
        onlySeeded: options.onlySeeded ? 'yes' : 'no'
      });

      const response = await fetch(`${baseURL}/search?${params}`);
      const data = await response.json();

      return data.results;
    } catch (err) {
      setError(err.message);
      return [];
    } finally {
      setLoading(false);
    }
  }, [baseURL]);

  const addTorrent = useCallback(async (torrentRaw) => {
    setLoading(true);
    setError(null);

    try {
      const response = await fetch(`${baseURL}/add`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ raw: torrentRaw })
      });

      const data = await response.json();
      return data.success;
    } catch (err) {
      setError(err.message);
      throw err;
    } finally {
      setLoading(false);
    }
  }, [baseURL]);

  const listDownloads = useCallback(async () => {
    setLoading(true);
    setError(null);

    try {
      const response = await fetch(`${baseURL}/list`);
      const data = await response.json();
      return data.items;
    } catch (err) {
      setError(err.message);
      return [];
    } finally {
      setLoading(false);
    }
  }, [baseURL]);

  return {
    loading,
    error,
    search,
    addTorrent,
    listDownloads
  };
}

// Uso en componente
function SearchComponent() {
  const { loading, error, search } = useJackent();
  const [results, setResults] = useState([]);
  const [query, setQuery] = useState('');

  const handleSearch = async () => {
    const torrents = await search(query, {
      trackers: '1337x',
      sort: 'seeders',
      onlySeeded: true
    });
    setResults(torrents);
  };

  return (
    <div>
      <input 
        value={query} 
        onChange={e => setQuery(e.target.value)}
        placeholder="Buscar torrents..."
      />
      <button onClick={handleSearch} disabled={loading}>
        {loading ? 'Buscando...' : 'Buscar'}
      </button>

      {error && <div className="error">{error}</div>}

      <div className="results">
        {results.map((torrent, i) => (
          <div key={i} className="torrent-card">
            <h3>{torrent.title}</h3>
            <p>{torrent.seeders} seeders | {torrent.size}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

export default SearchComponent;
